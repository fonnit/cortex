<!-- Item shape from Neon (prisma schema) — what /api/triage/GET returns -->
interface TriageItem {
  id: string
  source: 'gmail' | 'downloads'
  filename: string | null
  mime_type: string | null
  size_bytes: number | null
  source_metadata: {          // parsed from JSON column
    subject?: string          // email subject or filename
    from?: string
    received?: string
    snippet?: string
  } | null
  classification_trace: {
    stage1?: { reason?: string; suggestedRule?: string }
    stage2?: {
      proposals?: {
        type?: Array<{ value: string; conf: number }>
        from?: Array<{ value: string; conf: number }>
        context?: Array<{ value: string; conf: number }>
      }
      confident?: string[]    // axes above threshold: ['type', 'from']
    }
  } | null
  proposed_drive_path: string | null
  stage: 'relevance' | 'label'  // derived: if proposals exist → label, else → relevance
  ingested_at: string
}

<!-- Decision POST payload -->
interface TriageDecision {
  itemId: string
  type: 'keep' | 'ignore' | 'archive' | 'confirm' | 'skip'
  picks?: { Type?: string; From?: string; Context?: string }
}

<!-- Decision POST response -->
interface TriageDecisionResponse {
  ok: true
  status: 'uncertain' | 'certain' | 'ignored'
}

<!-- Key behavioral contracts from design/project/triage.jsx (copy verbatim) -->

Keyboard guards (ALL must be present):
  if (newOpen) return                                         // inline input open
  if (e.target.tagName === 'INPUT' || e.target.tagName === 'TEXTAREA') return
  if (e.metaKey || e.ctrlKey || e.altKey) return
  if (document.querySelector('[data-clerk-modal]')) return    // Clerk modal guard (ADDITIONAL — not in prototype)

Navigation:
  j → next item (Math.min(items.length-1, i+1))
  h → prev item (Math.max(0, i-1))

Relevance mode (item.stage === 'relevance'):
  k → keep,  x → ignore,  s → skip

Label mode (item.stage === 'label'):
  1/2/3 → pick proposal for first unresolved axis
  n → open new-category input for first unresolved axis
  a → archive as-is,  i → ignore,  s → skip,  Enter → confirm

Undo:
  u → restore lastAction.current.prev decided state
  Single-level only — lastAction.current stores { prev: decided }

useEffect on activeIdx: setPicks({}), setNewOpen(null), scrollIntoView({ block:'nearest', behavior:'smooth' })
useEffect on activeIdx: record openedAt = Date.now() for TRI-10 timing

Toast: showToast({ tag, subject }) — auto-dismisses after 2200ms
advance(): finds next undecided item after current; wraps to first undecided if none after

Langfuse decision timing (TRI-10):
  On each decision, emit langfuse.event({ traceId: item.id, name: 'triage.decision',
    metadata: { type: decision, durationMs: Date.now() - openedAt } })
  Use langfuse package from package.json (v3.38.20)
