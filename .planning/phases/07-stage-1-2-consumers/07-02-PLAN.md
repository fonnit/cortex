<objective>
Assemble the foundations from 07-01 into a runnable two-pool consumer process and ship the launchd plist
that boots it. Stage 1 worker drains the relevance queue with 10 concurrent claude invocations; Stage 2
worker drains the labelling queue with 2 concurrent invocations and fresh taxonomy per batch. Both emit
Langfuse traces chained off the API's X-Trace-Id, both POST `/api/classify` with success or error
outcomes, both treat a 409 stale-claim race as "give up cleanly" not "retry". The launchd plist runs
the process under `KeepAlive=true` so it auto-restarts on crash, and `ThrottleInterval=10` so it
backs off on persistent failures.

    ```ts
    import { z } from 'zod'
    import type Langfuse from 'langfuse'
    import { Semaphore } from './semaphore.js'
    import { invokeClaude } from './claude.js'
    import { buildStage1Prompt } from './prompts.js'
    import { getQueue, postClassify } from '../http/client.js'
    import type { QueueItem, ClassifyRequest } from '../http/types.js'

    export const STAGE1_LIMIT = 10
    export const STAGE1_CONCURRENCY = 10
    const POLL_INTERVAL_ITEMS_MS = 5_000
    const POLL_INTERVAL_EMPTY_MS = 30_000

    const Stage1ResultSchema = z.object({
      decision: z.enum(['keep','ignore','uncertain']),
      confidence: z.number().min(0).max(1),
      reason: z.string().optional(),
    })

    export interface Stage1Deps {
      langfuse: Langfuse
      // DI seams for testing — default to the real client functions.
      getQueueImpl?: typeof getQueue
      postClassifyImpl?: typeof postClassify
      invokeClaudeImpl?: typeof invokeClaude
    }

    export function runStage1Worker(deps: Stage1Deps): { stop: () => Promise<void> } {
      const sem = new Semaphore(STAGE1_CONCURRENCY)
      let stopped = false
      const inFlight = new Set<Promise<void>>()

      const handleOne = async (item: QueueItem, parentTraceId: string | null) => {
        const release = await sem.acquire()
        const span = deps.langfuse.trace({
          name: 'consumer-stage1-item',
          metadata: { item_id: item.id, source: item.source, inbound_trace_id: parentTraceId },
        })
        try {
          let prompt: string
          try {
            prompt = buildStage1Prompt(item)
          } catch (err) {
            await deps.postClassifyImpl!({
              item_id: item.id, stage: 1, outcome: 'error',
              error_message: `prompt_build_error: ${(err as Error).message}`,
            })
            return
          }
          const outcome = await deps.invokeClaudeImpl!(prompt, Stage1ResultSchema)
          let payload: ClassifyRequest
          if (outcome.kind === 'ok') {
            payload = { item_id: item.id, stage: 1, outcome: 'success',
              decision: outcome.value.decision, confidence: outcome.value.confidence,
              reason: outcome.value.reason }
          } else if (outcome.kind === 'parse_error') {
            payload = { item_id: item.id, stage: 1, outcome: 'error',
              error_message: `parse_error: ${outcome.reason}` }
          } else if (outcome.kind === 'exit_error') {
            payload = { item_id: item.id, stage: 1, outcome: 'error',
              error_message: `exit_error: code=${outcome.exitCode}` }
          } else {
            payload = { item_id: item.id, stage: 1, outcome: 'error',
              error_message: 'timeout' }
          }
          const result = await deps.postClassifyImpl!(payload)
          if (result.kind === 'conflict') {
            // Stale-claim race — log + move on. CONTEXT D-postClassify-no-retry-409.
            deps.langfuse.trace({ name:'consumer-stage1-conflict', metadata:{ item_id: item.id } })
          }
        } catch (err) {
          // Defensive — any unexpected throw must NOT crash the worker.
          deps.langfuse.trace({ name:'consumer-stage1-unexpected', metadata:{ item_id: item.id, error: String(err) } })
        } finally {
          release()
        }
      }

      const loop = async () => {
        while (!stopped) {
          const queueRes = await (deps.getQueueImpl ?? getQueue)({ stage: 1, limit: STAGE1_LIMIT })
          if ('kind' in queueRes && queueRes.kind === 'skip') {
            await sleep(POLL_INTERVAL_EMPTY_MS)
            continue
          }
          const r = queueRes as Exclude<typeof queueRes, { kind: 'skip' }>
          for (const item of r.items) {
            const p = handleOne(item, r.traceId).finally(() => inFlight.delete(p))
            inFlight.add(p)
          }
          await sleep(r.items.length > 0 ? POLL_INTERVAL_ITEMS_MS : POLL_INTERVAL_EMPTY_MS)
        }
      }
      const loopPromise = loop()

      return {
        async stop() {
          stopped = true
          await loopPromise.catch(() => {})
          await Promise.allSettled([...inFlight])
        },
      }
    }
    ```

    ```ts
    // agent/src/consumer/index.ts — Cortex consumer process entry point.
    // Phase 7 Plan 02 Task 3 (v1.1).
    //
    // SEPARATE process from agent/src/index.ts (the daemon). Started by
    // agent/launchd/com.cortex.consumer.plist. Runs Stage 1 (limit=10,
    // concurrency=10) and Stage 2 (limit=2, concurrency=2) worker loops in
    // parallel, draining /api/queue end-to-end.
    //
    // ZERO Neon access (HTTP only via /api/queue + /api/classify), ZERO Drive
    // calls. Two-pool architecture per CONTEXT D-process-layout.

    import Langfuse from 'langfuse'
    import { assertClaudeOnPath } from './claude.js'
    import { runStage1Worker } from './stage1.js'
    import { runStage2Worker } from './stage2.js'

    const REQUIRED_ENV = ['CORTEX_API_URL', 'CORTEX_API_KEY', 'LANGFUSE_PUBLIC_KEY', 'LANGFUSE_SECRET_KEY'] as const
    const SHUTDOWN_DRAIN_TIMEOUT_MS = 5_000

    export interface BootstrapEnvResult { ok: boolean; missing: string[] }

    export function validateConsumerEnv(): BootstrapEnvResult {
      const missing = REQUIRED_ENV.filter((k) => !process.env[k])
      return { ok: missing.length === 0, missing: [...missing] }
    }

    export interface BootstrapOpts {
      langfuse?: Langfuse
      runStage1?: typeof runStage1Worker
      runStage2?: typeof runStage2Worker
      assertClaudeOnPathImpl?: typeof assertClaudeOnPath
    }

    export async function bootstrapConsumer(opts?: BootstrapOpts): Promise<void> {
      const langfuse = opts?.langfuse ?? new Langfuse({
        publicKey: process.env.LANGFUSE_PUBLIC_KEY!,
        secretKey: process.env.LANGFUSE_SECRET_KEY!,
        baseUrl: process.env.LANGFUSE_HOST ?? 'https://cloud.langfuse.com',
        flushAt: 5,
        flushInterval: 30_000,
      })

      const envCheck = validateConsumerEnv()
      if (!envCheck.ok) {
        console.error(`[cortex-consumer] FATAL: missing required env: ${envCheck.missing.join(', ')}`)
        try {
          langfuse.trace({ name: 'consumer_bootstrap_fatal', metadata: { reason: 'missing_env', missing: envCheck.missing } })
          await langfuse.flushAsync()
        } catch { /* ignore */ }
        process.exit(1); return
      }

      try {
        await (opts?.assertClaudeOnPathImpl ?? assertClaudeOnPath)()
      } catch (err) {
        console.error(`[cortex-consumer] FATAL: ${(err as Error).message}`)
        try {
          langfuse.trace({ name: 'consumer_bootstrap_fatal', metadata: { reason: 'claude_cli_missing', error: String(err) } })
          await langfuse.flushAsync()
        } catch { /* ignore */ }
        process.exit(1); return
      }

      const stage1 = (opts?.runStage1 ?? runStage1Worker)({ langfuse })
      const stage2 = (opts?.runStage2 ?? runStage2Worker)({ langfuse })

      langfuse.trace({ name: 'consumer_start', metadata: { pid: process.pid, version: '0.1.1' } })
      console.log('[cortex-consumer] started (Stage 1 + Stage 2 pools running)')

      let shuttingDown = false
      const shutdown = async (): Promise<void> => {
        if (shuttingDown) return
        shuttingDown = true
        try {
          await Promise.race([
            Promise.all([stage1.stop(), stage2.stop()]),
            new Promise<void>((resolve) => setTimeout(resolve, SHUTDOWN_DRAIN_TIMEOUT_MS)),
          ])
        } catch { /* defensive */ }
        try { await langfuse.flushAsync() } catch { /* ignore */ }
      }

      const onSignal = (signal: NodeJS.Signals) => {
        void (async () => {
          try { await shutdown() } finally {
            const code = signal === 'SIGINT' ? 130 : 0
            process.exit(code)
          }
        })()
      }
      process.on('SIGTERM', () => onSignal('SIGTERM'))
      process.on('SIGINT',  () => onSignal('SIGINT'))
      process.on('uncaughtException', async (err) => {
        langfuse.trace({ name: 'consumer_uncaught_error', metadata: { error: err.message } })
        await shutdown()
        process.exit(1)
      })
    }

    const isTest = process.env.JEST_WORKER_ID !== undefined || process.env.NODE_ENV === 'test'
    if (!isTest) {
      bootstrapConsumer().catch((err) => {
        console.error('[cortex-consumer] bootstrap threw:', err)
        process.exit(1)
      })
    }
    ```
