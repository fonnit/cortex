    1. Create `scripts/acc/lib/argv-heuristics.mjs` exporting pure functions:

       ```js
       export const ARGV_SIZE_LIMIT = 16384 // 16KB — D-05

       export function parseArgvLine(line) {
         // Returns { pid: number, argv: string } | null
         // Match: leading WS, digits (pid), WS, rest-of-line (argv).
         // Skip the header line "PID COMMAND" and blank lines.
         const match = line.match(/^\s*(\d+)\s+(.+?)\s*$/)
         if (!match) return null
         return { pid: Number(match[1]), argv: match[2] }
       }

       export function isSuspiciousArgv(argv) {
         if (argv.length > ARGV_SIZE_LIMIT) {
           return { suspicious: true, reason: 'size', size: argv.length }
         }
         if (argv.includes('\0')) {
           return { suspicious: true, reason: 'null_byte' }
         }
         return { suspicious: false }
       }

       export function extractClaudeInvocations(lines) {
         // Match argv that begins with 'claude -p' OR includes ' node ... claude -p'
         // (since execFile('claude', ['-p', ...]) shows up as the basename or full path).
         // Returns Array<{ pid, argv }>.
         const re = /(?:^|\/)claude\s+-p\b/
         const out = []
         for (const line of lines) {
           const parsed = parseArgvLine(line)
           if (!parsed) continue
           if (re.test(parsed.argv)) out.push(parsed)
         }
         return out
       }
       ```

    4. Create `scripts/acc/audit-consumer-argv.sh`:

       ```bash
       #!/usr/bin/env bash
       # ACC-04 second half: during a live consumer cycle, sample ps -ww and
       # confirm no `claude -p` invocation has argv > 16KB or contains null
       # bytes (i.e. file content is never in argv).
       # Default --watch-for=60s. Polls every 1s.
       set -euo pipefail

       WATCH_SECONDS=60
       DRY_RUN=0
       for arg in "$@"; do
         case "$arg" in
           --watch-for=*) WATCH_SECONDS="${arg#--watch-for=}" ;;
           --dry-run)     DRY_RUN=1 ;;
           --help)
             echo "Usage: $0 [--watch-for=SECONDS] [--dry-run]"
             echo "  --watch-for=N: poll ps every 1s for N seconds (default 60)."
             echo "  Pass: every captured 'claude -p' argv ≤ 16KB AND no NUL bytes."
             exit 0
             ;;
           *) echo "Unknown arg: $arg" >&2; exit 2 ;;
         esac
       done

       if [[ $DRY_RUN -eq 1 ]]; then
         cat <<EOF
       [DRY-RUN] Would poll: ps -wwo pid,command
       [DRY-RUN] Watch duration: ${WATCH_SECONDS}s, interval 1s
       [DRY-RUN] Per-sample filter: argv begins with 'claude -p' (or '.../claude -p')
       [DRY-RUN] Per-invocation check: size ≤ 16384 AND no NUL byte (\\x00)
       [DRY-RUN] PASS = zero suspicious invocations across all samples
       [DRY-RUN] FAIL = any suspicious invocation
       [DRY-RUN] OK
       EOF
         exit 0
       fi

       NODE_BIN="$(command -v node)"
       if [[ -z "$NODE_BIN" ]]; then
         echo "FAIL: 'node' not on PATH (needed by argv-heuristics.mjs)" >&2
         exit 2
       fi

       SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
       SAMPLE_FILE="$(mktemp -t cortex-acc-argv.XXXXXX)"
       trap 'rm -f "$SAMPLE_FILE"' EXIT

       echo "Sampling for ${WATCH_SECONDS}s …"
       END=$(( $(date +%s) + WATCH_SECONDS ))
       SAMPLES=0
       INVOCATIONS=0
       while [[ $(date +%s) -lt $END ]]; do
         SAMPLES=$((SAMPLES+1))
         ps -wwo pid,command >> "$SAMPLE_FILE" || true
         echo "---SAMPLE---" >> "$SAMPLE_FILE"
         sleep 1
       done

       # Hand off to the pure-logic checker.
       RESULT="$("$NODE_BIN" "${SCRIPT_DIR}/lib/argv-heuristics.mjs" --check < "$SAMPLE_FILE" || true)"
       echo "$RESULT"
       if echo "$RESULT" | grep -q '^PASS '; then
         exit 0
       fi
       exit 1
       ```

       ```js
       export const REQUIRED_SPAN_NAMES = Object.freeze({
         apiIngest: 'api-ingest',
         apiQueue: 'api-queue',
         apiClassify: 'api-classify',
         stage1Item: 'consumer-stage1-item',
         stage2Item: 'consumer-stage2-item',
       })

       /**
        * @param {Array<{name: string, id: string, metadata?: Record<string, any>}>} traces
        * @param {{ requireStage2?: boolean }} [opts]  // requireStage2=true for keep items
        * @returns {{ ok: boolean, chain: string[], missing: string[], broken: string[] }}
        */
       export function walkSpanChain(traces, opts = {}) {
         const byName = new Map()
         for (const t of traces) {
           if (!byName.has(t.name)) byName.set(t.name, [])
           byName.get(t.name).push(t)
         }
         const have = (n) => byName.has(n) && byName.get(n).length > 0
         const required = [
           REQUIRED_SPAN_NAMES.apiIngest,
           REQUIRED_SPAN_NAMES.apiQueue,
           REQUIRED_SPAN_NAMES.apiClassify,
         ]
         const stage = opts.requireStage2
           ? REQUIRED_SPAN_NAMES.stage2Item
           : (have(REQUIRED_SPAN_NAMES.stage1Item)
               ? REQUIRED_SPAN_NAMES.stage1Item
               : REQUIRED_SPAN_NAMES.stage2Item)
         required.splice(2, 0, stage) // insert before classify

         const missing = required.filter((n) => !have(n))
         const chain = required.filter((n) => have(n))
         const broken = []

         // Verify stage* trace's metadata.inbound_trace_id resolves to a known api-queue trace.
         const queueTraces = byName.get(REQUIRED_SPAN_NAMES.apiQueue) ?? []
         const queueIds = new Set(queueTraces.map((t) => t.id))
         const stageTraces = byName.get(stage) ?? []
         for (const st of stageTraces) {
           const inbound = st.metadata?.inbound_trace_id
           if (!inbound) {
             broken.push(`${stage}->queue:no_inbound_trace_id`)
             continue
           }
           if (!queueIds.has(inbound)) {
             broken.push(`${stage}->queue:dangling_inbound`)
           }
         }

         return {
           ok: missing.length === 0 && broken.length === 0,
           chain,
           missing,
           broken,
         }
       }
       ```

       // ACC-05: reconstruct end-to-end span chain in Langfuse.
       // Usage: node scripts/acc/audit-langfuse-trace.mjs --item-id ITEM_ID
       //        node scripts/acc/audit-langfuse-trace.mjs --content-hash HASH
       //        node scripts/acc/audit-langfuse-trace.mjs --dry-run

       import { walkSpanChain, REQUIRED_SPAN_NAMES } from './lib/trace-walker.mjs'

       const args = parseArgs(process.argv.slice(2))
       if (args.help) { printHelp(); process.exit(0) }
       if (args.dryRun) {
         console.log('[DRY-RUN] Would fetch trace tree for the supplied item.')
         console.log('[DRY-RUN] Required span names:', Object.values(REQUIRED_SPAN_NAMES).join(', '))
         console.log('[DRY-RUN] Retry policy: 12 attempts × 5s backoff (60s total)')
         console.log('[DRY-RUN] OK')
         process.exit(0)
       }
       if (!args.itemId && !args.contentHash) {
         console.error('FAIL: must pass --item-id ID or --content-hash HASH (or --dry-run)')
         process.exit(2)
       }

       const publicKey = mustEnv('LANGFUSE_PUBLIC_KEY')
       const secretKey = mustEnv('LANGFUSE_SECRET_KEY')
       const baseUrl = process.env.LANGFUSE_HOST ?? 'https://cloud.langfuse.com'
       const lf = new Langfuse({ publicKey, secretKey, baseUrl })

       // Strategy: list traces filtered by the item's content_hash or item_id
       // tag/metadata field. The Langfuse SDK exposes lf.api.tracesList().
       // We list all traces with a recent time-window (default last 1h) and
       // collect those whose name matches our REQUIRED set AND whose metadata
       // (or input/output) references the item.
       const matchKey = args.itemId ? 'item_id' : 'content_hash'
       const matchVal = args.itemId ?? args.contentHash

       const RETRY_MAX = 12
       const RETRY_DELAY_MS = 5000
       let result = null
       for (let attempt = 1; attempt <= RETRY_MAX; attempt++) {
         const traces = await fetchRecentMatchingTraces(lf, matchKey, matchVal)
         result = walkSpanChain(traces, { requireStage2: !!args.requireStage2 })
         if (result.ok) break
         if (attempt < RETRY_MAX) {
           console.error(`Attempt ${attempt}/${RETRY_MAX}: missing=${result.missing.join(',')||'-'} broken=${result.broken.join(',')||'-'}; retrying in ${RETRY_DELAY_MS/1000}s …`)
           await sleep(RETRY_DELAY_MS)
         }
       }

       await lf.flushAsync()

       if (result?.ok) {
         console.log(`PASS ACC-05 (langfuse-trace) — chain: ${result.chain.join(' → ')}`)
         process.exit(0)
       }
       console.error('FAIL ACC-05 (langfuse-trace):')
       console.error('  missing spans:', result?.missing?.join(', ') || '-')
       console.error('  broken links:', result?.broken?.join(', ') || '-')
       process.exit(1)

       function parseArgs(argv) {
         const out = {}
         for (let i = 0; i < argv.length; i++) {
           const a = argv[i]
           if (a === '--help') out.help = true
           else if (a === '--dry-run') out.dryRun = true
           else if (a === '--require-stage2') out.requireStage2 = true
           else if (a === '--item-id') out.itemId = argv[++i]
           else if (a === '--content-hash') out.contentHash = argv[++i]
         }
         return out
       }
       function printHelp() {
         console.log('Usage: audit-langfuse-trace.mjs [--item-id ID | --content-hash HASH] [--require-stage2] [--dry-run]')
         console.log(`Required spans: ${Object.values(REQUIRED_SPAN_NAMES).join(', ')}`)
       }
       function mustEnv(name) {
         const v = process.env[name]
         if (!v) { console.error(`FAIL: missing env ${name}`); process.exit(2) }
         return v
       }
       function sleep(ms) { return new Promise((r) => setTimeout(r, ms)) }

       async function fetchRecentMatchingTraces(lf, matchKey, matchVal) {
         // List recent traces (last 1h) and filter by metadata match.
         // tracesList API: GET /api/public/traces with optional fromTimestamp.
         const fromTimestamp = new Date(Date.now() - 60 * 60 * 1000).toISOString()
         const resp = await lf.api.tracesList({ fromTimestamp, limit: 100 })
         const matching = []
         for (const summary of resp.data ?? []) {
           // Fetch full detail to inspect metadata / observations.
           try {
             const full = await lf.api.traceGet(summary.id)
             const md = full?.metadata ?? {}
             const inMetadata = md[matchKey] === matchVal
             const inInput = JSON.stringify(full?.input ?? '').includes(matchVal)
             if (inMetadata || inInput) {
               matching.push({ id: full.id, name: full.name, metadata: md })
             }
           } catch (err) {
             // Skip individual fetch failures.
           }
         }
         return matching
       }

       // ACC-02: rewind Gmail historyId to ~6 months ago, watch the queue

       // OR to look up an old historyId from a known message. This script

       // explicitly. --rewind=6mo prints guidance and exits 2.

       import path from 'node:path'
       import os from 'node:os'

         console.log('[DRY-RUN]    OR --history-id N (explicit rewind)')
         console.log('[DRY-RUN] After write: tail /tmp/cortex-consumer.log; exit when no Gmail items in queue for 5 consecutive minutes')
         console.log('[DRY-RUN] OK')
         process.exit(0)

       const stateDir = process.env.CORTEX_AGENT_STATE_DIR ?? path.join(os.homedir(), '.config', 'cortex')
       const cursorPath = path.join(stateDir, 'gmail-cursor.json')

       if (args.clear) {
         try {
           await unlink(cursorPath)
           console.log(`Cleared cursor at ${cursorPath}; next daemon poll will full-sync (ING-06).`)
         } catch (err) {
           if (err.code === 'ENOENT') {
             console.log(`No cursor at ${cursorPath} — already clear.`)
           } else { throw err }

         const { rename } = await import('node:fs/promises')
         await rename(tmp, cursorPath)
         console.log(`Wrote ${cursorPath} with last_history_id=${args.historyId}`)
       } else if (args.rewind === '6mo') {

         }

             if (err.code !== 'ENOENT') throw err
           }
           await new Promise((r) => setTimeout(r, 30_000))
