    **File 1: app/api/taxonomy/[axis]/[name]/route.ts**

    PATCH handler — accepts `{ op: 'rename' | 'deprecate', newName?: string }`:

    ```typescript
    import { requireAuth } from '@/lib/auth'
    import { prisma } from '@/lib/prisma'
    import { z } from 'zod'

    const PatchBody = z.discriminatedUnion('op', [
      z.object({ op: z.literal('rename'), newName: z.string().min(1).max(200) }),
      z.object({ op: z.literal('deprecate') }),
    ])

    export async function PATCH(
      req: Request,
      { params }: { params: Promise<{ axis: string; name: string }> }
    ) {
      try {
        const userId = await requireAuth()
        const { axis, name } = await params
        const decoded = decodeURIComponent(name)
        const body = PatchBody.parse(await req.json())

        const axisCol = axis === 'type' ? 'axis_type'
          : axis === 'from'    ? 'axis_from'
          : axis === 'context' ? 'axis_context'
          : null
        if (!axisCol) return Response.json({ error: 'invalid axis' }, { status: 400 })

        if (body.op === 'rename') {
          await prisma.$transaction([
            // Update all items that use this label value
            prisma.item.updateMany({
              where: { user_id: userId, [axisCol]: decoded },
              data: { [axisCol]: body.newName },
            }),
            // Rename the TaxonomyLabel row itself
            prisma.taxonomyLabel.update({
              where: { user_id_axis_name: { user_id: userId, axis, name: decoded } },
              data: { name: body.newName },
            }),
          ])
        } else {
          // deprecate
          await prisma.taxonomyLabel.update({
            where: { user_id_axis_name: { user_id: userId, axis, name: decoded } },
            data: { deprecated: true },
          })
        }

        return Response.json({ ok: true }, { headers: { 'Cache-Control': 'no-store' } })
      } catch (err) {
        if (err instanceof Response) return err
        console.error('[PATCH /api/taxonomy]', err)
        return Response.json({ error: 'Internal server error' }, { status: 500 })
      }
    }
    ```

    **File 2: app/api/taxonomy/merge/route.ts**

    POST handler — accepts `{ axis, sources: string[], canonical: string }`:

    - Validate with Zod: axis ∈ ['type','from','context'], sources is string[] min length 1, canonical is string min 1.
    - Map axis to axisCol same as above.
    - Run `prisma.$transaction`:
      1. `prisma.item.updateMany` — set `axisCol = canonical` where `axisCol IN sources` and `user_id = userId`.
      2. `prisma.taxonomyLabel.deleteMany` — delete TaxonomyLabel rows where `user_id = userId, axis, name IN sources, name != canonical`.
      3. `prisma.taxonomyMergeProposal.create` — `{ user_id: userId, axis, a: sources.join(', '), b: canonical, evidence: 'manual merge', suggested_canonical: canonical, status: 'accepted' }`.
    - Return `Response.json({ ok: true })`.
    - Error handling same pattern.
